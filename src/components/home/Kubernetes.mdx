```yaml
kind: Deployment
apiVersion: apps/v1
metadata:
  name: steep
spec:
  # cluster with 3 nodes enabling automatic
  # failover and zero downtime deployments
  replicas: 3

  selector:
    matchLabels:
      app: steep
  template:
    metadata:
      labels:
        app: steep
    spec:
      containers:
        - name: steep
          image: steep/steep
          env:
            - name: STEEP_HTTP_HOST
              value: "0.0.0.0"
          ports:
            - containerPort: 8080
```
